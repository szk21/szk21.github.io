	#include "Hex2Dec.h"

	int error_flag;

	int Hex2Dec(char *str_hex)
	{
		int i;
		int hex_head = 0;
		error_flag = 1;
		int res = 0;
		int tmp;

		//if str valid ?
		if (str_hex == NULL || *str_hex == '\0')
		{
			return res;
		}

		//if str start with "0x" ?
		int hex_len = strlen(str_hex);
		if (*str_hex == '0' && *(str_hex+1) == 'x')
		{
			hex_head = 2;
			str_hex += hex_head;
		}

		//if str_hex overflow ?
		if (hex_len - hex_head > 8)
		{
			return res;
		}

		while(*str_hex != '\0')
		{
			if (*str_hex >= '0' && *str_hex <= '9')
			{
				tmp = *str_hex - '0';
			}
			else if (*str_hex >= 'a' && *str_hex <= 'f')
			{
				tmp = *str_hex - 'a' + 10;
			}
			else if (*str_hex >= 'A' && *str_hex <= 'F')
			{
				tmp = *str_hex - 'A' + 10;
			}
			else
			{
				//invalid str
				res = 0;
				break;
			}

			str_hex++;
			//if the end ?
			if (*str_hex == '\0')
			{
				error_flag = 0;
			}
			res = res*16 + tmp;
		}

		return res;
	}
